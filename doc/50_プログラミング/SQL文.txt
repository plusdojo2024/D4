/* usersテーブルの作成 */
create table users (
users_id    integer         auto increment,
mail        varchar (100)   not null,
password    varchar (100)   not null,
grow_point  integer         default 0,
last_login_date date  not null
primary key (users_id)
);

/* questionsテーブルの作成 */
create table questions (
questions_id	integer		auto increment,
question		varchar (1000)	not null,
users_id		integer		not null,
judge		    varchar (10)	not null /* default ‘回答募集中’ */,
primary key(questions_id)
/* foreign key (users_id) references users(users_id) */
);

/* answersテーブルの作成 */
create table answers (
answers_id	integer		auto increment,
question_id	integer		not null,
answer		varchar (1000)	not null,
user_id		integer		not null,
primary key (answers_id)
/* foreign key (users_id) references users(users_id) */
);


/* requestsテーブルの作成 */
create table requests (
requests_id	    integer		    auto increment,
address_order 	varchar (20)	not null,
request		    varchar (1000)	not null,
users_id		integer		    not null,
primary key (requests_id)
/* foreign key (users_id) references users(users_id) */
);
